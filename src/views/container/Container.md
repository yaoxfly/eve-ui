  # Container
     用于布局的容器组件，置于中间内容区域，常与Main组件配合，方便快速搭建页面的基本结构,提供左右布局,上下布局，等布局方式
  # 基础用法
 <template>
  <div>
    <Example/>
  </div>
</template>
<script>
import Example from './Example'
import LeftRightExample from './LeftRightExample'
import UpDownExample from './UpDownExample'
export default {
  components: {
    Example,
    LeftRightExample,
    UpDownExample
  }
}
</script>
